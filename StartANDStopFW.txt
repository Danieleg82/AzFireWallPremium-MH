STOP:

# Stop an existing firewall

$azfw = Get-AzFirewall -Name "AzFW" -ResourceGroupName "AzFirewallPremiumTest"
$azfw.Deallocate()
Set-AzFirewall -AzureFirewall $azfw

START:

# Start the firewall

$azfw = Get-AzFirewall -Name "AzFW" -ResourceGroupName "AzFirewallPremiumTest"
$vnet = Get-AzVirtualNetwork -ResourceGroupName "AzFirewallPremiumTest" -Name "VNET1"
$publicip1 = Get-AzPublicIpAddress -Name "AzFWPIP" -ResourceGroupName "AzFirewallPremiumTest"
$azfw.Allocate($vnet,@($publicip1))

Set-AzFirewall -AzureFirewall $azfw
